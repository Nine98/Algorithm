#include<stdio.h>
int n;
int p[10];
void DFS(int index){
    if(index == n+1){
        for(int i=1;i<n+1;i++){
            cout<<p[i];
        }
        <<endl;
        return;
    }
    for(int i=0;i<n+1;i++){
        p[index]=i;
        DFS(index+1);
    }
}
int main(){
    n=3;
    DFS(1);
    return 1;
}
